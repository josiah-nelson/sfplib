#include <tunables/global>

profile sfpliberate flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities for Bluetooth and network operations
  capability net_admin,
  capability net_raw,
  capability dac_override,
  capability dac_read_search,
  capability setgid,
  capability setuid,

  # S6-Overlay v3 support
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /usr/lib/bashio/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/s6-overlay/** rwix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /run/s6/** rw,
  /var/run/s6/** rw,

  # Allow execution of Python and FastAPI
  /usr/bin/python3* ix,
  /usr/lib/python3*/** mr,
  /usr/local/lib/python3*/** mr,

  # Allow access to add-on files
  /app/** rwix,
  /usr/share/sfplib/** r,
  /usr/share/sfpliberate/** r,

  # Allow read/write to config directory (database, backups)
  /config/sfplib/** rw,
  /config/sfplib/ rw,
  /config/sfpliberate/** rw,
  /config/sfpliberate/ rw,

  # Allow read/write to share directory
  /share/** rw,

  # Allow read-only access to Home Assistant data
  /data/** r,

  # D-Bus access for Bluetooth
  #include <abstractions/dbus-strict>
  dbus (send, receive) bus=system,
  dbus (send, receive) bus=session,

  # Bluetooth device access
  /sys/class/bluetooth/** r,
  /sys/devices/** r,
  /dev/rfkill rw,

  # Network access for API
  network inet stream,
  network inet6 stream,
  network inet dgram,
  network inet6 dgram,
  network netlink raw,

  # Supervisor API access
  network unix stream,
  /run/supervisor.sock rw,

  # Temporary files
  /tmp/** rw,
  /var/tmp/** rw,

  # Proc and sys access
  /proc/*/net/dev r,
  /proc/sys/net/** r,
  /sys/kernel/mm/transparent_hugepage/hpage_pmd_size r,

  # DNS resolution
  /etc/resolv.conf r,
  /etc/hosts r,
  /etc/nsswitch.conf r,
  /etc/host.conf r,
  /run/systemd/resolve/stub-resolv.conf r,

  # SSL/TLS certificates
  /etc/ssl/openssl.cnf r,
  /etc/ssl/certs/** r,
  /usr/share/ca-certificates/** r,

  # Deny access to sensitive files
  deny /root/.ssh/** rw,
  deny /etc/shadow r,
  deny /etc/sudoers r,
}
